Basic If-Else Statements

Simple conditional logic in JavaScript

Static If-Else Examples

// Example 1: Simple if statement let temperature = 25; if (temperature > 20) { console.log("It's warm today!"); } // Example 2: If-else statement let age = 18; if (age >= 18) { console.log("You are an adult"); } else { console.log("You are a minor"); } // Example 3: Multiple conditions let score = 85; if (score >= 90) { console.log("Grade: A"); } else if (score >= 80) { console.log("Grade: B"); } else if (score >= 70) { console.log("Grade: C"); } else { console.log("Grade: F"); }

Interactive Age Checker

function checkAge(age) { if (age >= 65) { return "Senior citizen - Eligible for senior discounts!"; } else if (age >= 18) { return "Adult - You can vote and drive!"; } else if (age >= 13) { return "Teenager - Almost an adult!"; } else { return "Child - Enjoy your childhood!"; } }

Weather Condition Checker

function checkWeather(temperature, isRaining) { if (temperature > 30) { if (isRaining) { return "Hot and rainy - Stay indoors with AC!"; } else { return "Hot and sunny - Perfect for swimming!"; } } else if (temperature > 20) { if (isRaining) { return "Warm but rainy - Take an umbrella!"; } else { return "Pleasant weather - Great for outdoor activities!"; } } else { if (isRaining) { return "Cold and rainy - Stay warm and dry!"; } else { return "Cold but clear - Dress warmly!"; } } }

Number Comparison

function compareNumbers(num1, num2) { if (num1 > num2) { return `${num1} is greater than ${num2}`; } else if (num1 < num2) { return `${num1} is less than ${num2}`; } else { return `${num1} is equal to ${num2}`; } }

Password Strength Checker

function checkPasswordStrength(password) { if (password.length < 6) { return "Weak - Password too short!"; } else if (password.length < 8) { return "Fair - Consider making it longer!"; } else if (password.length < 12) { return "Good - Nice length!"; } else { return "Strong - Excellent password length!"; } }
