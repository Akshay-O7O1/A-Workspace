Break Statements

Using break to exit loops and switch statements in JavaScript

Break in For Loop

function findFirstEven(numbers) { let result = "No even number found"; for (let i = 0; i < numbers.length; i++) { console.log(`Checking: ${numbers[i]}`); if (numbers[i] % 2 === 0) { result = `First even number: ${numbers[i]} at index ${i}`; break; // Exit the loop when first even number is found } } return result; }

Break in While Loop

function guessNumber(target, maxAttempts) { let attempts = 0; let found = false; let log = []; while (attempts < maxAttempts) { attempts++; let guess = Math.floor(Math.random() * 10) + 1; log.push(`Attempt ${attempts}: Guessed ${guess}`); if (guess === target) { log.push(`üéâ Found the number ${target} in ${attempts} attempts!`); found = true; break; // Exit when number is found } } if (!found) { log.push(`‚ùå Could not find ${target} in ${maxAttempts} attempts`); } return log; }

Break in Switch Statement

function processCommand(command) { let result = ""; switch (command.toLowerCase()) { case 'start': result = "üöÄ System starting..."; break; // Prevents fall-through to next case case 'stop': result = "‚èπÔ∏è System stopping..."; break; case 'pause': result = "‚è∏Ô∏è System paused..."; break; case 'reset': result = "üîÑ System resetting..."; break; case 'status': result = "üìä System status: Running"; break; default: result = "‚ùì Unknown command"; } return result; }
Try commands: start, stop, pause, reset, status

Break with Nested Loops

function findInMatrix(matrix, target) { let found = false; let position = null; let log = []; outerLoop: // Label for outer loop for (let i = 0; i < matrix.length; i++) { for (let j = 0; j < matrix[i].length; j++) { log.push(`Checking position [${i}][${j}]: ${matrix[i][j]}`); if (matrix[i][j] === target) { position = [i, j]; found = true; log.push(`‚úÖ Found ${target} at position [${i}][${j}]`); break outerLoop; // Break out of both loops } } } if (!found) { log.push(`‚ùå ${target} not found in matrix`); } return { found, position, log }; }

Break in String Processing

function findFirstVowel(text) { const vowels = ['a', 'e', 'i', 'o', 'u']; let log = []; for (let i = 0; i < text.length; i++) { let char = text[i].toLowerCase(); log.push(`Checking character '${text[i]}' at position ${i}`); if (vowels.includes(char)) { log.push(`üéØ First vowel '${text[i]}' found at position ${i}`); return { vowel: text[i], position: i, log }; } } log.push("‚ùå No vowels found in the text"); return { vowel: null, position: -1, log }; }

Break with User Input Validation

function validatePasswords(passwords) { let log = []; let validPassword = null; for (let i = 0; i < passwords.length; i++) { let password = passwords[i]; log.push(`Validating password ${i + 1}: "${password}"`); if (password.length < 6) { log.push(`‚ùå Too short (${password.length} characters)`); continue; } if (!/[A-Z]/.test(password)) { log.push(`‚ùå No uppercase letter`); continue; } if (!/[0-9]/.test(password)) { log.push(`‚ùå No number`); continue; } // If we reach here, password is valid log.push(`‚úÖ Valid password found!`); validPassword = password; break; // Stop checking once we find a valid password } if (!validPassword) { log.push(`‚ùå No valid passwords found`); } return { validPassword, log }; }
Password requirements: At least 6 characters, 1 uppercase letter, 1 number
